23 double haversine(const std::pair<double, double> &t_a, const std::pair<double, double> &t_b) {
24 auto [lat1, lon1] = t_a;
25 auto [lat2, lon2] = t_b;
26 // distance between latitudes
27 // and longitudes
28 const double dLat = (lat2 - lat1) * M_PI / 180.0;
29 const double dLon = (lon2 - lon1) * M_PI / 180.0;
30
31 // convert to radians
32 lat1 = (lat1) * M_PI / 180.0;
33 lat2 = (lat2) * M_PI / 180.0;
34
35 // apply formulae
36 const double a =

39 const double rad = 3958.755866;
40 const double c = 2 * std::asin(std::sqrt(a));
