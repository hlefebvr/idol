52template<class T>
53double idol::Point<T>::objective_value() const {
54
55 if (m_objective_value.has_value()) {
56 return m_objective_value.value();
57 }
58
59 if (m_status == Unbounded) {
60 return -Inf;
61 }
62
63 return Inf;
64}
