\begin{equation*}
    \min_{x\in X} \ c^\top x + \max_{ u\in U } \ \min_{y\in Y(x,u)} \ d^\top y

\begin{equation*}
    X \define \Defset{ x\in \tilde{X} }{ Ax \ge a }
\end{equation*}

\begin{equation*}
    Y(x,u) \define \Defset{ y\in \tilde{Y} }{ Cx + Dy + E(x)u \ge b }
\end{equation*}

\begin{equation*}
    U \define \Defset{ u\in\tilde{U} }{ Fu \le g }
\end{equation*}
